    # Likelihood
      for(i in 1:N){
        y[i] ~ dbern(mu[i])
        logit(mu[i]) <- alpha[year[i]] + beta*flow[i]
      }

    # Priors
      for(i in 1:ngroups){
        alpha[i] ~ dnorm(mu.int, tau.int)
      }

      tau.int <- 1/(sigma.int*sigma.int)
      sigma.int ~ dunif(0, 100)
      beta ~ dnorm(0, 0.001)     

  }
