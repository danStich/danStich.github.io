                 
  model {                       
    # Likelihood
      for(i in 1:nobs){    
        N[i] ~ dnegbin(p[i], r[color[i]]) 
        logit(p[i]) <- alpha[color[i]] + beta[color[i]]*mass[i]
      }

    # Priors
      for(i in 1:ncolors){
        alpha[i] ~ dnorm(0,0.0001) 
        beta[i] ~ dnorm(0,0.0001)
        r[i] ~ dgamma(0.0001, 0.0001)T(0,3)
      }
  }
